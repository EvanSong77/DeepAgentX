```python
# models/skill.py

from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from enum import Enum
from datetime import datetime

class SkillStatus(Enum):
    """技能状态"""
    ACTIVE = "active"
    INACTIVE = "inactive"
    LOADING = "loading"
    ERROR = "error"
    DEPRECATED = "deprecated"

@dataclass
class Tool:
    """工具定义"""
    name: str
    description: str
    parameters: Dict[str, Any]  # JSON Schema
    required_params: List[str] = field(default_factory=list)
    return_type: str = "string"
    tags: List[str] = field(default_factory=list)
    
    def to_openai_tool_format(self) -> Dict[str, Any]:
        """转换为 OpenAI Tool 格式"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters
            }
        }

@dataclass
class Skill:
    """技能定义"""
    # 基本信息
    id: str
    name: str
    version: str
    description: str
    category: str
    
    # 工具列表
    tools: List[Tool]
    
    # 元数据
    author: str
    license: str = "MIT"
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    
    # 依赖和配置
    dependencies: List[str] = field(default_factory=list)
    config_schema: Optional[Dict[str, Any]] = None
    
    # 路径和状态
    base_path: str = ""  # 技能文件所在目录
    status: SkillStatus = SkillStatus.ACTIVE
    error_message: Optional[str] = None
    
    # 使用统计
    usage_count: int = 0
    last_used_at: Optional[datetime] = None
    success_rate: float = 1.0
    
    def is_available(self) -> bool:
        """检查技能是否可用"""
        return self.status == SkillStatus.ACTIVE
    
    def get_tool_by_name(self, tool_name: str) -> Optional[Tool]:
        """通过名称获取工具"""
        for tool in self.tools:
            if tool.name == tool_name:
                return tool
        return None
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "id": self.id,
            "name": self.name,
            "version": self.version,
            "description": self.description,
            "category": self.category,
            "tools": [
                {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.parameters
                }
                for tool in self.tools
            ],
            "author": self.author,
            "status": self.status.value,
        }
```

```python
# models/execution.py

from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from enum import Enum
from datetime import datetime

class ExecutionStatus(Enum):
    """执行状态"""
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"
    PAUSED = "paused"

class ToolCallStatus(Enum):
    """工具调用状态"""
    QUEUED = "queued"
    EXECUTING = "executing"
    SUCCESS = "success"
    FAILED = "failed"
    TIMEOUT = "timeout"

@dataclass
class ToolCall:
    """单个工具调用"""
    id: str
    tool_name: str
    skill_name: str
    parameters: Dict[str, Any]
    status: ToolCallStatus = ToolCallStatus.QUEUED
    
    # 执行信息
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None
    duration_ms: float = 0.0
    
    # 结果
    result: Any = None
    error: Optional[str] = None
    error_type: Optional[str] = None
    
    # 元数据
    attempt: int = 1
    max_retries: int = 3
    tags: List[str] = field(default_factory=list)
    
    def get_duration(self) -> float:
        """获取执行耗时（毫秒）"""
        if self.started_at and self.completed_at:
            return (self.completed_at - self.started_at).total_seconds() * 1000
        return 0.0
    
    def is_successful(self) -> bool:
        """判断是否成功"""
        return self.status == ToolCallStatus.SUCCESS
    
    def should_retry(self) -> bool:
        """判断是否应该重试"""
        return (
            self.status == ToolCallStatus.FAILED and
            self.attempt < self.max_retries
        )

@dataclass
class ExecutionFrame:
    """执行栈帧"""
    id: str
    parent_id: Optional[str]
    execution_id: str
    
    # 执行内容
    tool_calls: List[ToolCall] = field(default_factory=list)
    llm_messages: List[Dict[str, Any]] = field(default_factory=list)
    
    # 状态
    status: ExecutionStatus = ExecutionStatus.PENDING
    depth: int = 0
    
    # 时间信息
    started_at: datetime = field(default_factory=datetime.now)
    completed_at: Optional[datetime] = None

@dataclass
class Execution:
    """完整的执行记录"""
    # 基本信息
    id: str
    session_id: str
    user_message: str
    
    # 执行栈
    frames: List[ExecutionFrame] = field(default_factory=list)
    root_frame: Optional[ExecutionFrame] = None
    
    # 状态
    status: ExecutionStatus = ExecutionStatus.PENDING
    
    # 时间信息
    started_at: datetime = field(default_factory=datetime.now)
    completed_at: Optional[datetime] = None
    total_duration_ms: float = 0.0
    
    # 结果
    final_response: Optional[str] = None
    
    # 统计信息
    total_tool_calls: int = 0
    successful_tool_calls: int = 0
    failed_tool_calls: int = 0
    
    def add_frame(self, frame: ExecutionFrame) -> None:
        """添加执行栈帧"""
        self.frames.append(frame)
        if not self.root_frame:
            self.root_frame = frame
    
    def get_frame_tree(self) -> Dict[str, Any]:
        """获取执行栈树结构"""
        if not self.root_frame:
            return {}
        return self._build_tree(self.root_frame)
    
    def _build_tree(self, frame: ExecutionFrame) -> Dict[str, Any]:
        """递归构建树"""
        children = [
            self._build_tree(f) for f in self.frames
            if f.parent_id == frame.id
        ]
        return {
            "id": frame.id,
            "tool_calls": [
                {
                    "name": tc.tool_name,
                    "status": tc.status.value,
                    "duration": tc.get_duration()
                }
                for tc in frame.tool_calls
            ],
            "children": children
        }
    
    def get_execution_summary(self) -> Dict[str, Any]:
        """获取执行摘要"""
        return {
            "id": self.id,
            "status": self.status.value,
            "total_duration_ms": self.total_duration_ms,
            "total_tool_calls": self.total_tool_calls,
            "successful_calls": self.successful_tool_calls,
            "failed_calls": self.failed_tool_calls,
            "success_rate": (
                self.successful_tool_calls / self.total_tool_calls
                if self.total_tool_calls > 0 else 1.0
            )
        }
```

```python
# models/message.py

from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from enum import Enum
from datetime import datetime

class MessageRole(Enum):
    """消息角色"""
    USER = "user"
    ASSISTANT = "assistant"
    TOOL = "tool"
    SYSTEM = "system"

@dataclass
class Message:
    """消息"""
    id: str
    role: MessageRole
    content: str
    
    # 元数据
    timestamp: datetime = field(default_factory=datetime.now)
    session_id: str = ""
    execution_id: Optional[str] = None
    
    # 附加信息
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    # Tool 相关
    tool_name: Optional[str] = None
    tool_call_id: Optional[str] = None
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "role": self.role.value,
            "content": self.content,
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'Message':
        """从字典创建"""
        return cls(
            id=data.get("id", ""),
            role=MessageRole(data["role"]),
            content=data["content"],
        )

@dataclass
class Conversation:
    """对话"""
    id: str
    session_id: str
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    
    # 消息列表
    messages: List[Message] = field(default_factory=list)
    
    # 统计信息
    total_messages: int = 0
    user_messages: int = 0
    assistant_messages: int = 0
    
    def add_message(self, message: Message) -> None:
        """添加消息"""
        self.messages.append(message)
        self.total_messages += 1
        
        if message.role == MessageRole.USER:
            self.user_messages += 1
        elif message.role == MessageRole.ASSISTANT:
            self.assistant_messages += 1
        
        self.updated_at = datetime.now()
    
    def get_context(self, max_messages: int = 10) -> List[Dict[str, str]]:
        """获取对话上下文"""
        messages = self.messages[-max_messages:]
        return [msg.to_dict() for msg in messages]
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "id": self.id,
            "created_at": self.created_at.isoformat(),
            "total_messages": self.total_messages,
            "messages": [msg.to_dict() for msg in self.messages]
        }
```

```python
# core/skill_manager.py

import json
import os
from pathlib import Path
from typing import List, Dict, Any, Optional
from concurrent.futures import ThreadPoolExecutor
import logging

from models.skill import Skill, Tool, SkillStatus
from utils.validators import validate_skill_json, validate_tool_parameters
from utils.logger import get_logger

logger = get_logger(__name__)

class SkillManager:
    """技能管理器 - 负责技能的发现、加载、索引和管理"""
    
    def __init__(self, skills_dir: str = "./skills"):
        """
        初始化技能管理器
        
        Args:
            skills_dir: 技能目录路径
        """
        self.skills_dir = Path(skills_dir)
        self.skills_dir.mkdir(parents=True, exist_ok=True)
        
        self.skills: Dict[str, Skill] = {}
        self.skill_index: Dict[str, str] = {}  # name -> id mapping
        self.tools_index: Dict[str, str] = {}  # tool_name -> skill_id mapping
        
        self._load_lock = threading.Lock()
    
    def discover_skills(self) -> List[str]:
        """
        自动发现所有技能
        
        Returns:
            发现的技能 ID 列表
        """
        logger.info(f"发现技能目录: {self.skills_dir}")
        discovered_skills = []
        
        # 遍历技能目录
        for skill_dir in self.skills_dir.iterdir():
            if not skill_dir.is_dir():
                continue
            
            skill_json_path = skill_dir / "skill.json"
            if not skill_json_path.exists():
                logger.warning(f"技能目录 {skill_dir.name} 没有 skill.json")
                continue
            
            try:
                skill_id = self._load_skill(skill_dir)
                discovered_skills.append(skill_id)
                logger.info(f"✓ 发现技能: {skill_id}")
            except Exception as e:
                logger.error(f"✗ 加载技能失败 {skill_dir.name}: {e}")
        
        logger.info(f"总共发现 {len(discovered_skills)} 个技能")
        return discovered_skills
    
    def _load_skill(self, skill_dir: Path) -> str:
        """
        加载单个技能
        
        Args:
            skill_dir: 技能目录路径
            
        Returns:
            技能 ID
        """
        skill_json_path = skill_dir / "skill.json"
        
        # 读取 JSON 文件
        with open(skill_json_path, 'r', encoding='utf-8') as f:
            skill_data = json.load(f)
        
        # 验证 JSON 格式
        validate_skill_json(skill_data)
        
        # 构建 Skill 对象
        skill = Skill(
            id=skill_data.get("id", skill_data.get("name")),
            name=skill_data["name"],
            version=skill_data.get("version", "1.0.0"),
            description=skill_data.get("description", ""),
            category=skill_data.get("category", "general"),
            tools=[
                Tool(
                    name=t["name"],
                    description=t.get("description", ""),
                    parameters=t.get("parameters", {
                        "type": "object",
                        "properties": {}
                    }),
                    required_params=t.get("required", []),
                    return_type=t.get("return_type", "string"),
                    tags=t.get("tags", [])
                )
                for t in skill_data.get("tools", [])
            ],
            author=skill_data.get("author", "unknown"),
            license=skill_data.get("license", "MIT"),
            dependencies=skill_data.get("dependencies", []),
            config_schema=skill_data.get("config_schema"),
            base_path=str(skill_dir),
            status=SkillStatus.ACTIVE,
        )
        
        # 加载配置文件（如果存在）
        config_path = skill_dir / "config.yaml"
        if config_path.exists():
            skill._config_path = str(config_path)
        
        # 加载实现文件（如果存在）
        tools_path = skill_dir / "tools.py"
        if tools_path.exists():
            skill._tools_path = str(tools_path)
        
        # 存储到注册表
        self.skills[skill.id] = skill
        self.skill_index[skill.name] = skill.id
        
        # 更新工具索引
        for tool in skill.tools:
            self.tools_index[tool.name] = skill.id
        
        return skill.id
    
    def reload_skills(self) -> None:
        """重新加载所有技能"""
        with self._load_lock:
            self.skills.clear()
            self.skill_index.clear()
            self.tools_index.clear()
            self.discover_skills()
    
    def get_skill(self, skill_id: str) -> Optional[Skill]:
        """获取技能"""
        return self.skills.get(skill_id)
    
    def get_skill_by_name(self, skill_name: str) -> Optional[Skill]:
        """通过名称获取技能"""
        skill_id = self.skill_index.get(skill_name)
        return self.skills.get(skill_id) if skill_id else None
    
    def get_all_skills(self) -> List[Skill]:
        """获取所有技能"""
        return list(self.skills.values())
    
    def get_available_skills(self) -> List[Skill]:
        """获取所有可用的技能"""
        return [s for s in self.skills.values() if s.is_available()]
    
    def get_skill_by_tool_name(self, tool_name: str) -> Optional[Skill]:
        """通过工具名称获取技能"""
        skill_id = self.tools_index.get(tool_name)
        return self.skills.get(skill_id) if skill_id else None
    
    def get_skills_by_category(self, category: str) -> List[Skill]:
        """通过分类获取技能"""
        return [
            s for s in self.skills.values()
            if s.category == category
        ]
    
    def get_tools_for_agent(self) -> List[Dict[str, Any]]:
        """获取供 Agent 使用的工具列表（OpenAI 格式）"""
        tools = []
        for skill in self.get_available_skills():
            for tool in skill.tools:
                tools.append({
                    "type": "function",
                    "function": {
                        "name": tool.name,
                        "description": f"[{skill.name}] {tool.description}",
                        "parameters": tool.parameters,
                        "metadata": {
                            "skill_id": skill.id,
                            "skill_name": skill.name,
                        }
                    }
                })
        return tools
    
    def validate_tool_call(self, tool_name: str, parameters: Dict[str, Any]) -> bool:
        """
        验证工具调用
        
        Args:
            tool_name: 工具名称
            parameters: 参数
            
        Returns:
            是否有效
        """
        skill = self.get_skill_by_tool_name(tool_name)
        if not skill:
            return False
        
        tool = skill.get_tool_by_name(tool_name)
        if not tool:
            return False
        
        # 验证参数
        try:
            validate_tool_parameters(tool.parameters, parameters)
            return True
        except ValueError:
            return False
    
    def get_skill_stats(self) -> Dict[str, Any]:
        """获取技能统计信息"""
        skills = self.get_all_skills()
        active_skills = [s for s in skills if s.is_available()]
        
        return {
            "total_skills": len(skills),
            "active_skills": len(active_skills),
            "total_tools": sum(len(s.tools) for s in skills),
            "categories": list(set(s.category for s in skills)),
            "by_category": {
                category: len(self.get_skills_by_category(category))
                for category in set(s.category for s in skills)
            }
        }
```

## 3. Tool Handler 实现

```python
# core/tool_handler.py

import importlib.util
import sys
from typing import List, Dict, Any, Tuple
import asyncio
from concurrent.futures import ThreadPoolExecutor
import logging

from models.skill import Tool
from models.execution import ToolCall, ToolCallStatus
from utils.logger import get_logger
from .skill_manager import SkillManager

logger = get_logger(__name__)

class ToolHandler:
    """工具处理器 - 负责工具的执行、结果处理和错误恢复"""
    
    def __init__(self, skill_manager: SkillManager):
        """
        初始化工具处理器
        
        Args:
            skill_manager: 技能管理器实例
        """
        self.skill_manager = skill_manager
        self.tool_cache = {}  # 工具实现的缓存
        self.executor = ThreadPoolExecutor(max_workers=10)
    
    async def execute_tool(
        self,
        tool_name: str,
        parameters: Dict[str, Any],
        timeout: float = 30.0
    ) -> Tuple[bool, Any]:
        """
        执行单个工具
        
        Args:
            tool_name: 工具名称
            parameters: 参数
            timeout: 超时时间（秒）
            
        Returns:
            (成功标志, 结果或错误)
        """
        try:
            # 获取技能和工具
            skill = self.skill_manager.get_skill_by_tool_name(tool_name)
            if not skill:
                return False, f"技能未找到，工具: {tool_name}"
            
            tool = skill.get_tool_by_name(tool_name)
            if not tool:
                return False, f"工具未找到: {tool_name}"
            
            # 加载工具实现
            tool_func = self._load_tool_implementation(skill, tool_name)
            if not tool_func:
                return False, f"无法加载工具实现: {tool_name}"
            
            # 执行工具（带超时）
            loop = asyncio.get_event_loop()
            result = await asyncio.wait_for(
                loop.run_in_executor(
                    self.executor,
                    tool_func,
                    parameters
                ),
                timeout=timeout
            )
            
            logger.info(f"✓ 工具执行成功: {tool_name}")
            return True, result
        
        except asyncio.TimeoutError:
            logger.error(f"✗ 工具执行超时: {tool_name}")
            return False, f"工具执行超时（{timeout}s）: {tool_name}"
        
        except Exception as e:
            logger.error(f"✗ 工具执行错误: {tool_name}: {e}")
            return False, f"工具执行错误: {str(e)}"
    
    async def execute_tool_calls(
        self,
        tool_calls: List[Dict[str, Any]],
        parallel: bool = True
    ) -> List[Tuple[str, bool, Any]]:
        """
        执行多个工具调用
        
        Args:
            tool_calls: 工具调用列表
            parallel: 是否并行执行
            
        Returns:
            [(tool_name, success, result), ...]
        """
        results = []
        
        if parallel and len(tool_calls) > 1:
            # 并行执行
            tasks = [
                self.execute_tool(call["name"], call.get("parameters", {}))
                for call in tool_calls
            ]
            execution_results = await asyncio.gather(*tasks)
            
            for tool_call, (success, result) in zip(tool_calls, execution_results):
                results.append((tool_call["name"], success, result))
        else:
            # 顺序执行
            for tool_call in tool_calls:
                success, result = await self.execute_tool(
                    tool_call["name"],
                    tool_call.get("parameters", {})
                )
                results.append((tool_call["name"], success, result))
        
        return results
    
    def _load_tool_implementation(self, skill, tool_name: str) -> Any:
        """
        动态加载工具实现
        
        Args:
            skill: Skill 对象
            tool_name: 工具名称
            
        Returns:
            工具函数或 None
        """
        # 检查缓存
        cache_key = f"{skill.id}:{tool_name}"
        if cache_key in self.tool_cache:
            return self.tool_cache[cache_key]
        
        # 加载实现文件
        tools_path = getattr(skill, '_tools_path', None)
        if not tools_path:
            return None
        
        try:
            # 动态导入模块
            spec = importlib.util.spec_from_file_location(
                f"skill_{skill.id}_tools",
                tools_path
            )
            module = importlib.util.module_from_spec(spec)
            sys.modules[spec.name] = module
            spec.loader.exec_module(module)
            
            # 获取工具函数
            tool_func = getattr(module, tool_name, None)
            
            if tool_func:
                self.tool_cache[cache_key] = tool_func
                return tool_func
            else:
                logger.warning(
                    f"工具函数未在模块中找到: {tool_name} in {tools_path}"
                )
                return None
        
        except Exception as e:
            logger.error(f"加载工具实现失败: {e}")
            return None
    
    def validate_tool_call(
        self,
        tool_name: str,
        parameters: Dict[str, Any]
    ) -> Tuple[bool, str]:
        """
        验证工具调用
        
        Args:
            tool_name: 工具名称
            parameters: 参数
            
        Returns:
            (是否有效, 错误信息)
        """
        # 检查工具是否存在
        skill = self.skill_manager.get_skill_by_tool_name(tool_name)
        if not skill:
            return False, f"技能未找到: {tool_name}"
        
        tool = skill.get_tool_by_name(tool_name)
        if not tool:
            return False, f"工具未找到: {tool_name}"
        
        # 验证参数
        try:
            from utils.validators import validate_tool_parameters
            validate_tool_parameters(tool.parameters, parameters)
            return True, ""
        except ValueError as e:
            return False, str(e)
```

```python
# core/execution_tracker.py

from typing import List, Dict, Any, Optional
from datetime import datetime
import json
import logging

from models.execution import Execution, ExecutionFrame, ExecutionStatus, ToolCall
from utils.logger import get_logger

logger = get_logger(__name__)

class ExecutionTracker:
    """执行追踪器 - 记录和展示完整的执行链路"""
    
    def __init__(self, storage=None):
        """
        初始化执行追踪器
        
        Args:
            storage: 存储后端（可选）
        """
        self.current_execution: Optional[Execution] = None
        self.execution_history: Dict[str, Execution] = {}
        self.storage = storage
    
    def start_execution(
        self,
        execution_id: str,
        session_id: str,
        user_message: str
    ) -> Execution:
        """
        开始新的执行
        
        Args:
            execution_id: 执行 ID
            session_id: 会话 ID
            user_message: 用户消息
            
        Returns:
            Execution 对象
        """
        execution = Execution(
            id=execution_id,
            session_id=session_id,
            user_message=user_message,
        )
        
        # 创建根执行栈帧
        root_frame = ExecutionFrame(
            id=f"{execution_id}-frame-0",
            parent_id=None,
            execution_id=execution_id,
            depth=0,
        )
        execution.add_frame(root_frame)
        execution.root_frame = root_frame
        
        self.current_execution = execution
        self.execution_history[execution_id] = execution
        
        logger.info(f"开始执行: {execution_id}")
        return execution
    
    def record_tool_call(
        self,
        tool_name: str,
        skill_name: str,
        parameters: Dict[str, Any],
        tool_call_id: str
    ) -> ToolCall:
        """
        记录工具调用
        
        Args:
            tool_name: 工具名称
            skill_name: 技能名称
            parameters: 参数
            tool_call_id: 工具调用 ID
            
        Returns:
            ToolCall 对象
        """
        if not self.current_execution:
            raise RuntimeError("没有活跃的执行")
        
        tool_call = ToolCall(
            id=tool_call_id,
            tool_name=tool_name,
            skill_name=skill_name,
            parameters=parameters,
        )
        
        # 添加到当前栈帧
        if self.current_execution.root_frame:
            self.current_execution.root_frame.tool_calls.append(tool_call)
        
        self.current_execution.total_tool_calls += 1
        
        logger.debug(f"记录工具调用: {tool_name}")
        return tool_call
    
    def record_tool_call_result(
        self,
        tool_call_id: str,
        success: bool,
        result: Any,
        error: Optional[str] = None,
        duration_ms: float = 0.0
    ) -> None:
        """
        记录工具调用结果
        
        Args:
            tool_call_id: 工具调用 ID
            success: 是否成功
            result: 结果
            error: 错误信息
            duration_ms: 耗时（毫秒）
        """
        if not self.current_execution:
            raise RuntimeError("没有活跃的执行")
        
        # 查找工具调用
        tool_call = None
        for frame in self.current_execution.frames:
            for tc in frame.tool_calls:
                if tc.id == tool_call_id:
                    tool_call = tc
                    break
        
        if not tool_call:
            logger.warning(f"工具调用未找到: {tool_call_id}")
            return
        
        # 更新工具调用
        tool_call.completed_at = datetime.now()
        tool_call.duration_ms = duration_ms
        tool_call.result = result
        tool_call.error = error
        
        if success:
            tool_call.status = ToolCallStatus.SUCCESS
            self.current_execution.successful_tool_calls += 1
        else:
            tool_call.status = ToolCallStatus.FAILED
            self.current_execution.failed_tool_calls += 1
        
        logger.debug(f"记录工具结果: {tool_call.tool_name} - {success}")
    
    def complete_execution(self, final_response: str) -> Execution:
        """
        完成执行
        
        Args:
            final_response: 最终响应
            
        Returns:
            完成的 Execution 对象
        """
        if not self.current_execution:
            raise RuntimeError("没有活跃的执行")
        
        execution = self.current_execution
        execution.status = ExecutionStatus.COMPLETED
        execution.final_response = final_response
        execution.completed_at = datetime.now()
        
        # 计算总耗时
        if execution.started_at and execution.completed_at:
            execution.total_duration_ms = (
                (execution.completed_at - execution.started_at).total_seconds() * 1000
            )
        
        logger.info(
            f"完成执行: {execution.id} "
            f"({execution.total_duration_ms:.0f}ms, "
            f"{execution.successful_tool_calls}/{execution.total_tool_calls} 成功)"
        )
        
        # 保存到存储
        if self.storage:
            self.storage.save_execution(execution)
        
        self.current_execution = None
        return execution
    
    def get_execution(self, execution_id: str) -> Optional[Execution]:
        """获取执行记录"""
        return self.execution_history.get(execution_id)
    
    def get_execution_tree(self, execution_id: str) -> Optional[Dict[str, Any]]:
        """获取执行树"""
        execution = self.get_execution(execution_id)
        if not execution:
            return None
        return execution.get_frame_tree()
    
    def get_execution_summary(self, execution_id: str) -> Optional[Dict[str, Any]]:
        """获取执行摘要"""
        execution = self.get_execution(execution_id)
        if not execution:
            return None
        return execution.get_execution_summary()
    
    def get_recent_executions(self, limit: int = 10) -> List[Execution]:
        """获取最近的执行"""
        return sorted(
            self.execution_history.values(),
            key=lambda e: e.started_at,
            reverse=True
        )[:limit]
    
    def export_execution_as_json(self, execution_id: str) -> Optional[str]:
        """导出执行记录为 JSON"""
        execution = self.get_execution(execution_id)
        if not execution:
            return None
        
        # TODO: 实现完整的 JSON 导出逻辑
```

```python
    def shutdown(self) -> None:
        """关闭所有服务"""
        if hasattr(self.tool_handler, 'executor'):
            self.tool_handler.executor.shutdown(wait=True)
```
