from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from datetime import datetime
from enum import Enum

class StatusEnum(Enum):
    """状态枚举"""
    ACTIVE = "active"
    INACTIVE = "inactive"
    ERROR = "error"

@dataclass
class ExampleModel:
    """
    示例数据模型
    
    这是一个详细的类文档字符串，说明：
    - 类的目的
    - 主要属性
    - 使用示例
    """
    
    # 基本信息
    id: str
    name: str
    description: str = ""
    
    # 可选字段带默认值
    status: StatusEnum = StatusEnum.ACTIVE
    created_at: datetime = field(default_factory=datetime.now)
    metadata: Dict[str, Any] = field(default_factory=dict)
    items: List[str] = field(default_factory=list)
    
    def validate(self) -> bool:
        """验证数据的有效性"""
        if not self.id or not self.name:
            return False
        return True
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典表示"""
        return {
            "id": self.id,
            "name": self.name,
            "status": self.status.value,
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'ExampleModel':
        """从字典创建实例"""
        return cls(
            id=data["id"],
            name=data["name"],
            description=data.get("description", ""),
        )

### 3.2 函数设计规范

```python
async def process_tool_call(
    tool_name: str,
    parameters: Dict[str, Any],
    timeout: float = 30.0,
    retry_count: int = 3
) -> Tuple[bool, Any]:
    """
    处理工具调用
    
    这个函数的功能是：
    1. 验证工具是否存在
    2. 验证参数的有效性
    3. 执行工具（带超时和重试）
    4. 返回结果或错误
    
    Args:
        tool_name: 工具名称
        parameters: 工具参数字典
        timeout: 执行超时时间（秒）
        retry_count: 重试次数
        
    Returns:
        (成功标志, 结果或错误信息)
        
    Raises:
        ToolNotFoundError: 工具不存在
        ParameterValidationError: 参数无效
        ExecutionTimeoutError: 执行超时
        
    Example:
        >>> success, result = await process_tool_call(
        ...     "search",
        ...     {"query": "python tutorial"}
        ... )
        >>> if success:
        ...     print(f"Result: {result}")
        ... else:
        ...     print(f"Error: {result}")
    """
    logger.debug(f"处理工具调用: {tool_name}")
    
    # 验证阶段
    if not _validate_tool_exists(tool_name):
        logger.error(f"工具不存在: {tool_name}")
        raise ToolNotFoundError(f"Tool not found: {tool_name}")
    
    if not _validate_parameters(tool_name, parameters):
        logger.error(f"参数无效: {tool_name}")
        raise ParameterValidationError(f"Invalid parameters for {tool_name}")
    
    # 执行阶段（带重试）
    last_error = None
    for attempt in range(1, retry_count + 1):
        try:
            logger.debug(f"执行工具 (尝试 {attempt}/{retry_count}): {tool_name}")
            
            result = await asyncio.wait_for(
                _execute_tool(tool_name, parameters),
                timeout=timeout
            )
            
            logger.info(f"✓ 工具执行成功: {tool_name}")
            return True, result
        
        except asyncio.TimeoutError as e:
            last_error = ExecutionTimeoutError(f"Tool execution timeout: {tool_name}")
            logger.warning(f"⏱ 工具执行超时: {tool_name} (尝试 {attempt})")
        
        except Exception as e:
            last_error = e
            logger.warning(f"⚠ 工具执行失败: {tool_name} (尝试 {attempt}): {e}")
        
        # 指数退避重试
        if attempt < retry_count:
            wait_time = 2 ** (attempt - 1)
            await asyncio.sleep(wait_time)
    
    logger.error(f"✗ 工具最终执行失败: {tool_name}")
    return False, str(last_error)
```

### 3.5 类型注解规范

```python
from typing import List, Dict, Any, Optional, Union, Callable, TypeVar

T = TypeVar('T')

def process_items(
    items: List[str],
    processor: Callable[[str], str],
    skip_empty: bool = True
) -> Dict[str, str]:
    """
    处理项目列表
    
    Args:
        items: 字符串列表
        processor: 处理函数
        skip_empty: 是否跳过空字符串
        
    Returns:
        处理结果字典
    """
    result = {}
    for item in items:
        if skip_empty and not item:
            continue
        result[item] = processor(item)
    return result

def get_or_default(
    data: Dict[str, Any],
    key: str,
    default: T
) -> Union[Any, T]:
    """获取字典值，返回默认值"""
    return data.get(key, default)
```
